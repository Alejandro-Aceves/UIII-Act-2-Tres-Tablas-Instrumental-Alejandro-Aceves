```python). 
# app_instrumental/views.py

from django.shortcuts import render, redirect, get_object_or_404
from django.http import HttpResponse
from .models import Instrumento
from datetime import date 

# Vista de inicio (pantalla principal)
def inicio_instrumental(request): 
    # Simplemente renderiza la plantilla inicio.html
    context = {
        'fecha_sistema': date.today().strftime("%Y-%m-%d"),
    }
    return render(request, 'inicio.html', context)

# -----------------
# [cite_start]CRUD para INSTRUMENTOS [cite: 64]
# -----------------

# [C] - CREATE: Muestra el formulario para agregar
def agregar_instrumento(request): 
    if request.method == 'POST':
        # [cite_start]No validar entrada de datos [cite: 92]
        nombre_val = request.POST.get('nombre')
        marca_val = request.POST.get('marca')
        modelo_val = request.POST.get('modelo')
        precio_venta_val = request.POST.get('precio_venta')
        cantidad_stock_val = request.POST.get('cantidad_stock')
        categoria_val = request.POST.get('categoria')
        fecha_ingreso_val = request.POST.get('fecha_ingreso', date.today()) 

        # Crea y guarda el nuevo objeto
        Instrumento.objects.create(
            nombre=nombre_val,
            marca=marca_val,
            modelo=modelo_val,
            precio_venta=precio_venta_val,
            cantidad_stock=cantidad_stock_val,
            categoria=categoria_val,
            fecha_ingreso=fecha_ingreso_val
        )
        return redirect('ver_instrumento') # Redirige a la tabla de instrumentos

    return render(request, 'instrumentos/agregar_instrumento.html')


# [R] - READ: Muestra la lista de instrumentos en una tabla
def ver_instrumento(request): 
    instrumentos = Instrumento.objects.all() # Obtiene todos los registros
    context = {
        'instrumentos': instrumentos,
    }
    return render(request, 'instrumentos/ver_instrumento.html', context)


# [U] - UPDATE: Muestra el formulario con datos precargados
def actualizar_instrumento(request, id): 
    # Obtiene el objeto Instrumento o devuelve un 404
    instrumento = get_object_or_404(Instrumento, id_instrumento=id)
    context = {
        'instrumento': instrumento,
    }
    return render(request, 'instrumentos/actualizar_instrumento.html', context)

# [U] - UPDATE: Procesa el formulario de actualización
def realizar_actualizacion_instrumento(request): 
    if request.method == 'POST':
        # Obtiene el ID del instrumento a actualizar
        id_instrumento_val = request.POST.get('id_instrumento')
        instrumento = get_object_or_404(Instrumento, id_instrumento=id_instrumento_val)
        
        # Actualiza los campos
        instrumento.nombre = request.POST.get('nombre')
        instrumento.marca = request.POST.get('marca')
        instrumento.modelo = request.POST.get('modelo')
        instrumento.precio_venta = request.POST.get('precio_venta')
        instrumento.cantidad_stock = request.POST.get('cantidad_stock')
        instrumento.categoria = request.POST.get('categoria')
        instrumento.fecha_ingreso = request.POST.get('fecha_ingreso')
        
        instrumento.save() # Guarda los cambios
        
        return redirect('ver_instrumento') # Redirige a la tabla

    return redirect('ver_instrumento')


# [D] - DELETE: Muestra la confirmación para borrar
def borrar_instrumento(request, id): 
    instrumento = get_object_or_404(Instrumento, id_instrumento=id)
    
    if request.method == 'POST':
        instrumento.delete()
        return redirect('ver_instrumento')

    context = {
        'instrumento': instrumento,
    }
    return render(request, 'instrumentos/borrar_instrumento.html', context)
```
